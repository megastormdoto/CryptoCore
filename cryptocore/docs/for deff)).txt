python -c "
from Crypto.Cipher import AES
from Crypto.Util.Padding import pad, unpad
import binascii

key = binascii.unhexlify('000102030405060708090A0B0C0D0E0F')
iv = binascii.unhexlify('000102030405060708090A0B0C0D0E0F')

with open('megastorm_test.txt', 'rb') as f:
    data = f.read()

cipher = AES.new(key, AES.MODE_CBC, iv)
encrypted = cipher.encrypt(pad(data, 16))
with open('megastorm_test.enc', 'wb') as f:
    f.write(encrypted)

decipher = AES.new(key, AES.MODE_CBC, iv)
decrypted = unpad(decipher.decrypt(encrypted), 16)
with open('megastorm_test.dec', 'wb') as f:
    f.write(decrypted)

if data == decrypted:
    print('Тест прошел, все успешно')
else:
    print('Ошибка')
"
