openssl enc -aes-128-cbc -d -in megadoto.enc -out megadoto_decrypted.txt -K 000102030405060708090A0B0C0D0E0F -iv 000102030405060708090A0B0C0D0E0F

diff megadoto.txt megadoto_decrypted.txt && echo "шифрование и дешифрование CBC работает через OpenSSL"
