```python
# src/mac/hmac.py
class HMAC:
    """HMAC implementation from scratch following RFC 2104"""
    
    def __init__(self, key: bytes, hash_func: str = 'sha256'):
        # –ü–æ–¥–≥–æ—Ç–æ–≤–∫–∞ –∫–ª—é—á–∞
        self.key = self._prepare_key(key)
        self.hash_func = hash_func
        
    def _prepare_key(self, key: bytes) -> bytes:
        # –ï—Å–ª–∏ –∫–ª—é—á –¥–ª–∏–Ω–Ω–µ–µ –±–ª–æ–∫–∞ - —Ö—ç—à–∏—Ä—É–µ–º –µ–≥–æ
        # –ï—Å–ª–∏ –∫–æ—Ä–æ—á–µ - –¥–æ–ø–æ–ª–Ω—è–µ–º –Ω—É–ª—è–º–∏
        block_size = 64  # –¥–ª—è SHA-256
        if len(key) > block_size:
            if self.hash_func == 'sha256':
                key = SHA256().hash(key)
            else:  # sha3-256
                key = hashlib.sha3_256(key).digest()
        elif len(key) < block_size:
            key = key + b'\x00' * (block_size - len(key))
        return key
    
    def compute(self, message: bytes) -> bytes:
        # HMAC –≤—ã—á–∏—Å–ª–µ–Ω–∏–µ: H((K ‚äï opad) || H((K ‚äï ipad) || message))
        block_size = 64
        opad = bytes([0x5C] * block_size)
        ipad = bytes([0x36] * block_size)
        
        k_ipad = bytes(a ^ b for a, b in zip(self.key, ipad))
        k_opad = bytes(a ^ b for a, b in zip(self.key, opad))
        
        if self.hash_func == 'sha256':
            inner_hash = SHA256().hash(k_ipad + message)
            return SHA256().hash(k_opad + inner_hash)
        else:  # sha3-256
            inner_hash = hashlib.sha3_256(k_ipad + message).digest()
            return hashlib.sha3_256(k_opad + inner_hash).digest()
```
